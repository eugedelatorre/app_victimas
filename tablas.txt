TABLAS:

instituciones:

CREATE  TABLE instituciones  (id_instituciones int  unsigned not null auto_increment primary key , nombre_referencia varchar(255) , delitos int, descripcion_caso varchar(255), fecha_ingreso datetime, modalidad_ingreso int, organismos int, cual_otro_organismo varchar(255), cavaj int, fiscalia_juzgado varchar(255), causa_id_judicial int, comisaria varchar(255), denuncias_previas int, departamento_judicial int, estado int, motivospasivos int, cual_otro_motivospasivo varchar(255));

personas_asistidas:

CREATE  TABLE personas_asistidas  (nombre_persona_asistida varchar(255) , vinculo_persona_asistida int, otro_vinculo_persona_asistida_cual varchar(255), telefono_persona_asistida int, domicilio_persona_asistida varchar(255), localidad_persona_asistida int);

profesionales_intervinientes:

CREATE TABLE profesionales_intervinientes  (id_profesionales_intervinientes int  unsigned not null auto_increment primary key ,nombre_profesional_interviniente varchar(255) , desde_profesional_interviniente datetime, actual_profesional_interviniente int, hasta_profesional_interviniente datetime, idInstitucion int)
