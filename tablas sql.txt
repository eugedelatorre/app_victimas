

CREATE TABLE instituciones(id_instituciones int  unsigned not null auto_increment primary key,nombre_referencia varchar(255) , delitos int, descripcion_caso varchar(255), fecha_ingreso datetime, modalidad_ingreso int, cavaj int, fiscalia_juzgado varchar(255), causa_id_judicial varchar(255), comisaria varchar(255), departamento_judicial int, estado int);

CREATE  TABLE profesionales_intervinientes(id_profesionales_intervinientes int  unsigned not null auto_increment primary key,nombre_profesional_interviniente varchar(255) , desde_profesional_interviniente datetime, actual_profesional_interviniente int, hasta_profesional_interviniente datetime);
